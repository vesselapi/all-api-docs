```python Python
import http.client

conn = http.client.HTTPSConnection("api.vessel.land")

headers = { 'Content-Type': "application/json" }

conn.request("GET", "/connection/list", headers=headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
```
