```python Python
import http.client

conn = http.client.HTTPSConnection("api.vessel.land")

payload = "{\n  \"accessToken\": \"string\"\n}"

headers = {
    'Content-Type': "application/json",
    'vessel-api-token': ""
    }

conn.request("POST", "/connection/remove", payload, headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
```
