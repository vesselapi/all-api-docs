import http.client

conn = http.client.HTTPSConnection("api.vessel.dev")

payload = "{\n  \"connectionId\": \"string\"\n}"

headers = {
    'Content-Type': "application/json",
    'vessel-api-token': ""
    }

conn.request("DELETE", "/connection/connection", payload, headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
