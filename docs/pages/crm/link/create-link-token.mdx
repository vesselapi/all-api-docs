```js JavaScript Fetch
  },

    console.log(response);
  })
  .catch((err) => {

  });
```

```python Python
import http.client

conn = http.client.HTTPSConnection("api.vessel.dev")

payload = ""

headers = {
    'Content-Type': "application/json",
    'x-vessel-api-token': ""
    }

conn.request("POST", "/link/token", payload, headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
```
