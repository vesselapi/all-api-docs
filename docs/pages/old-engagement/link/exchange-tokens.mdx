```js JavaScript Fetch
fetch("https://api.vessel.land/link/exchange", {

  },

  });
```

```python Python
import http.client

conn = http.client.HTTPSConnection("api.vessel.land")

payload = "{\n  \"publicToken\": \"string\"\n}"

headers = {
    'Content-Type': "application/json",
    'vessel-api-token': ""
    }

conn.request("POST", "/link/exchange", payload, headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
```
